#include <iostream>
#include <vector>
using namespace std;
int main(){
    int a,b;
    vector<int>v[200];
    for(int i = 0;i<50;i++){
        cin>>a>>b;
        v[a].push_back(b);//array index is independent variable
    }
    for(int i = 0;i<=199;i++){
        int sz = v[i].size();
        if(sz>0){
            cout<<i<<":   ";
        }
        for(int j : v[i]){
            cout<<j<<",";
        }
        if(sz>0){
            cout<<"\n";
        }
    }
    return 0;
}
